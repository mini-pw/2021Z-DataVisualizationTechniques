library(ggplot2)
data <- data.frame(
  ilo = c(16.9,16.0,14.9,10.7,7.5,6.6,5.2,5.2,4.5,3.3,2.9,2.6,2.0,1.6),
  opis = c("Pozostae","Ubezpieczenia spoeczne","Subwencje oglne dla JST","Obrona narodowa","Program Rodzina 500+","Obsuga dugu Skarbu Pastwa","Skadka do Unii Europejskiej","Nauka i szkolnictwo wysze","Bezpieczestwo publiczne i ochrona przeciwpoarowa","Wymiar sprawiedliwoci","Transport i czno","Ochrona zdrowia","Rolnictwo","Wspfinansowanie projektw z udzialem rodkw z UE")
)

ggplot(data,aes(x = ilo, y = factor(opis,level=opis))) + 
  geom_bar(stat='identity')+
  ylab("Sektor budetu pastwa")+
  xlab("Kwota w zotych")+
  xlim(0,18)+
  labs(title="Twoje 100 z w budecie pastwa")+
  geom_text(aes(label = ilo),vjust = 0, hjust = 0)+
  theme_minimal()
